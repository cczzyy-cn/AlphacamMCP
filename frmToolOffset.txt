Private Declare PtrSafe Function FindWindow Lib "user32" Alias "FindWindowA" (ByVal lpClassName As String, ByVal lpWindowName As String) As Long
Private Declare PtrSafe Function GetWindowLong Lib "user32" Alias "GetWindowLongA" (ByVal hWnd As Long, ByVal nIndex As Long) As Long
Private Declare PtrSafe Function SetWindowLong Lib "user32" Alias "SetWindowLongA" (ByVal hWnd As Long, ByVal nIndex As Long, ByVal dwNewLong As Long) As Long
Private Declare PtrSafe Function SetWindowPos Lib "user32" (ByVal hWnd As Long, ByVal hWndInsertAfter As Long, ByVal X As Long, ByVal Y As Long, ByVal cx As Long, ByVal cy As Long, ByVal wFlags As Long) As Long
Private Const GWL_STYLE As Long = (-16)
Private Const WS_MINIMIZEBOX As Long = &H20000
Private Const SWP_NOSIZE As Long = &H1
Private Const SWP_NOMOVE As Long = &H2
Private Const SWP_NOZORDER As Long = &H4
Private Const SWP_FRAMECHANGED As Long = &H20
Private Sub AddMinButton(frm As Object)
    Dim hWnd As Long
    hWnd = FindWindow("ThunderDFrame", frm.Caption)
    If hWnd = 0 Then hWnd = FindWindow("ThunderRT5Form", frm.Caption)
    If hWnd = 0 Then Exit Sub
    Dim style As Long
    style = GetWindowLong(hWnd, GWL_STYLE)
    style = style Or WS_MINIMIZEBOX
    SetWindowLong hWnd, GWL_STYLE, style
    SetWindowPos hWnd, 0, 0, 0, 0, 0, SWP_NOMOVE Or SWP_NOSIZE Or SWP_NOZORDER Or SWP_FRAMECHANGED
End Sub
Private Sub UserForm_Activate(): AddMinButton Me: End Sub
Private Sub UserForm_Initialize()
    Me.Caption = "全排版刀具偏移"
    Dim drw As Drawing: Set drw = App.ActiveDrawing
    If drw Is Nothing Then Exit Sub
    Dim ops As Operations: Set ops = drw.Operations
    If ops Is Nothing Or ops.count = 0 Then Exit Sub
    Dim i As Long, j As Long
    For i = 1 To ops.count
        Dim op As Operation: Set op = ops(i)
        Dim subs As SubOperations: Set subs = op.SubOperations
        If Not (subs Is Nothing) Then
            For j = 1 To subs.count
                Dim subop As SubOperation: Set subop = subs(j)
                Dim t As MillTool: Set t = subop.Tool
                If Not (t Is Nothing) Then
                    Dim procName As String: procName = subop.Name
                    Dim spacePos As Integer
                    spacePos = InStr(procName, "  ")
                    If spacePos > 0 Then procName = Left(procName, spacePos - 1) Else: spacePos = InStr(procName, " "): If spacePos > 0 Then procName = Left(procName, spacePos - 1)
                    Dim displayText As String: displayText = procName & " - T" & CStr(t.Number) & " " & t.Name
                    Dim found As Boolean: found = False
                    Dim k As Long
                    For k = 0 To cmbTool.ListCount - 1
                        If cmbTool.List(k) = displayText Then found = True: Exit For
                    Next k
                    If Not found Then cmbTool.AddItem displayText
                End If
            Next j
        End If
    Next i
    If cmbTool.ListCount > 0 Then cmbTool.ListIndex = 0
    txtX.Text = "0": txtY.Text = "0": txtZ.Text = "0"
End Sub
Private Sub cmdOK_Click()
    If cmbTool.ListIndex < 0 Then MsgBox "请选择一把刀具！", vbExclamation: Exit Sub
    Dim xVal As Double, yVal As Double, zVal As Double
    xVal = Val(txtX.Text): yVal = Val(txtY.Text): zVal = Val(txtZ.Text)
    If xVal = 0 And yVal = 0 And zVal = 0 Then MsgBox "偏移量不能全为 0！", vbExclamation: Exit Sub
    Dim toolName As String
    Dim dashPos As Integer: dashPos = InStr(cmbTool.Text, " - ")
    If dashPos > 0 Then
        ' 提取 " - T# 刀具名" 中的刀具名
        Dim afterDash As String: afterDash = Mid(cmbTool.Text, dashPos + 3)
        ' 跳过 "T#" 前缀（T + 数字 + 空格）
        Dim skipT As Integer: skipT = InStr(afterDash, " ")
        If skipT > 0 Then
            toolName = Mid(afterDash, skipT + 1)
        Else
            toolName = afterDash
        End If
        ' 去除首尾空格
        toolName = Trim(toolName)
    Else
        toolName = Trim(cmbTool.Text)
    End If
    Dim drw As Drawing: Set drw = App.ActiveDrawing
    If drw Is Nothing Then Exit Sub
    App.SetUndoCommandName "全排版刀具偏移"
    App.SetUndoPoint
    drw.ScreenUpdating = False
    Dim count As Long
    Dim ops As Operations: Set ops = drw.Operations
    Dim i As Long, j As Long
    
    ' 方式1：通过 Operations 遍历
    For i = 1 To ops.count
        Dim op As Operation: Set op = ops(i)
        Dim subs As SubOperations: Set subs = op.SubOperations
        If Not (subs Is Nothing) Then
            For j = 1 To subs.count
                Dim subop As SubOperation: Set subop = subs(j)
                Dim t As MillTool: Set t = subop.Tool
                If Not (t Is Nothing) Then
                    ' 匹配：刀具名精确匹配 → 包含匹配 → T 号匹配
                    Dim matchTool As Boolean: matchTool = False
                    If t.Name = toolName Then
                        matchTool = True
                    ElseIf toolName <> "" And InStr(1, t.Name, toolName, vbTextCompare) > 0 Then
                        matchTool = True
                    ElseIf toolName <> "" And InStr(1, CStr(t.Number), toolName, vbTextCompare) > 0 Then
                        matchTool = True
                    End If
                    If matchTool Then
                        Dim tps As Paths: Set tps = subop.ToolPaths
                        If Not (tps Is Nothing) Then
                            Dim m As Long
                            For m = 1 To tps.count
                                Dim tp As Path: Set tp = tps(m)
                                If Not (tp Is Nothing) Then tp.MoveG xVal, yVal, zVal: count = count + 1
                            Next m
                        End If
                    End If
                End If
            Next j
        End If
    Next i
        ' Fallback: direct toolpath when no operations
    If count = 0 Then
        Dim tpIdx2 As Long
        Dim tpCnt2 As Long: tpCnt2 = drw.GetToolPathCount
        If tpCnt2 > 0 Then
            Dim tpP2 As Path: Set tpP2 = drw.GetFirstToolPath
            For tpIdx2 = 1 To tpCnt2
                If Not (tpP2 Is Nothing) Then
                    Dim t2 As MillTool: Set t2 = tpP2.GetTool
                    If Not (t2 Is Nothing) Then
                        If t2.Name = toolName Or (toolName <> "" And (InStr(1, t2.Name, toolName, vbTextCompare) > 0 Or InStr(1, CStr(t2.Number), toolName, vbTextCompare) > 0)) Then
                            tpP2.MoveG xVal, yVal, zVal: count = count + 1
                        End If
                    End If
                    Set tpP2 = tpP2.GetNext
                End If
            Next tpIdx2
        End If
    End If

drw.ScreenUpdating = True: drw.Redraw
    MsgBox "偏移完成！已处理 " & count & " 条刀具路径。" & vbCrLf & "刀具: " & toolName & vbCrLf & "偏移量: X=" & xVal & "  Y=" & yVal & "  Z=" & zVal, vbInformation, "全排版刀具偏移"
    Me.Show
End Sub
Private Sub cmdCancel_Click(): Unload Me: End Sub
